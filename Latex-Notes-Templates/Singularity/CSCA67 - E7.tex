\begin{question}{2.1}
    Prove that $\sqrt{2}$ is irrational.
\end{question}

\setcounter{equation}{0}
\begin{flalign}
    &\text{Suppose } \sqrt{2} \text{ is rational} && \ \text{for contradiction} \\
    &\quad\exists a\exists b, (\sqrt{2} = \frac{a}{b}) \land (a, b \text{ have no common factors} \ne 1.) && \ \text{(1, Definition of $\mathbb{Q}$)}\\
    &\quad(\sqrt{2} = \frac{x}{y}) \land (x, y \text{ have no common factors} \ne 1.) && \ \text{(2, E.I)}\\
    &\quad x^2 =  2y^2 && \ \text{(3, simp. algebra)} \\
    &\quad y^2 \text{ is an integer} && \\
    &\quad \exists k, x^2 = 2k && \ \text{(4, 5, E.G)} \\
    &\quad x^2 \text{ is even} && \ \text{(6, Definition of even)} \\ 
    &\quad x \text{ is even} && \ \text{(7, def., U.M.P)} \\ 
    &\quad x = 2i && \ \text{(8, def. E.I)} \\
    &\quad y^2 = \frac{x^2}{2} = \frac{(2i)^2}{2} = 2i^2 && \ \text{(4, 9)}\\
    &\quad y^2 \text{ is even} && \ \text{(10, def.)} \\
    &\quad y  \text{ is even} && \ \text{(11, def., U.M.P)} \\
    &\quad y = 2j && \ \text{12, def. E.I} \\
    &\quad 2 \text{ is a common factor of} x, y && \ \text{(9, 13)} \\
    &\quad (a, b \text{ have no common factors } \ne 1) && \ \text{(3, simp.)} \\
    &\quad \text{a contradiction}&& \ \text{(14, 15)} \\
    &\sqrt{2} \text{ is irrational.}&& \ \text{(1, 16, contradiction.)}
\end{flalign}
\begin{center}
    As required to prove. $\blacksquare$
\end{center}

\begin{question}{2.2}
    Prove that the sum of a rational number and an irrational number is irrational.
\end{question}
\[
    WTS: \forall x \forall y, (x \in \mathbb{Q} \land y \notin \mathbb{Q}) \rightarrow ((x+y) \notin \mathbb{Q})
\]

\setcounter{equation}{0}
\begin{flalign}
    &\text{Let } x, y \text{ be arbitrary} && \\
    &\quad\text{Suppose } (x \in \mathbb{Q}) \land (y \notin \mathbb{Q}) &&  \\
    &\quad\quad\text{Suppose } (x + y) \in \mathbb{Q} && \text{(for contradiction)} \\
    &\quad\quad\quad x \in Q && \text{(2, simp.)} \\
    &\quad\quad\quad y \notin Q && \text{(2, simp.)} \\
    &\quad\quad\quad \exists a, b \in \mathbb{Z}, ((b \ne 0) \land (x = \frac{a}{b})) && \text{(4, def$^n$ of $\mathbb{Q}$)} \\
    &\quad\quad\quad ((b \ne 0) \land (x = \frac{a}{b})) && \text{(6, E.I.)} \\
    &\quad\quad\quad x = \frac{a}{b} && \text{(7, simp.)} \\
    &\quad\quad\quad \exists u, v \in \mathbb{Z}, ((v \ne 0) \land (x + y = \frac{u}{v})) && \text{(3, def$^n$ of $\mathbb{Q}$)} \\
    &\quad\quad\quad ((v \ne 0) \land (x+y = \frac{u}{v})) && \text{(9, E.I.)} \\
    &\quad\quad\quad x+y = \frac{u}{v} && \text{(10, simp.)} \\
    &\quad\quad\quad y = (x + y) - x && \text{(math)} \\
    &\quad\quad\quad y = \frac{bu - au}{vb} && \text{(8, 11, math)} \\
    &\quad\quad\quad v \ne 0 && \text{(7, simp.)} \\
    &\quad\quad\quad b \ne 0 && \text{(10, simp.)} \\
    &\quad\quad\quad vb \ne 0 && \text{(14, 15, math)} \\
    &\quad\quad\quad (vb \ne 0) \land (y = \frac{bu - au}{vb}) && \text{(16, 13, conj.)} \\
    &\quad\quad\quad \exists c, d \in \mathbb{Z}, (d \ne 0) \land (y = \frac{c}{d}) && \text{(17, E.G)} \\
    &\quad\quad\quad y \in \mathbb{Q} && \text{(18, def$^n$ of $\mathbb{Q}$)} \\
    &\quad\quad\quad \text{a contradiction} && \text{(5, 19)} \\
    &\quad\quad x + y \notin \mathbb{Q} && \text{(3, 20, contradiction)} \\
    &\quad(x \in \mathbb{Q} \land y \notin \mathbb{Q}) \rightarrow (x+y \notin \mathbb{Q}) && \text{(2, 21, imp.)}\\
    &\forall x \forall y, (x \in \mathbb{Q} \land y \notin \mathbb{Q}) \rightarrow (x+y \notin \mathbb{Q}) && \text{(1, 22, U.G)}
\end{flalign}
\begin{center}
    As required to prove. $\blacksquare$
\end{center}
